¿Qué es Docker Compose?

  • Docker compose es una herramienta que permite definir y ejecutar múltiples contenedores de Docker.
  • Cada contenedor de Docker, es lo que se conoce como servicio dentro de Docker Compose.
  • Por defecto el archivo se llama docker-compose.yml, aunque puede especificarse para utilizar otro.

Estructura básica del docker-compose.yml


            version: "3"

            services:

            volumes:

            networks:

Definiendo servicios


            version: "3"
            services:
              app:
                build: . 
                ports:
                  - "3000:3000"
                  - "5000:5000"
                links:
                  - redis:redisAlias
              redis:
                image: redis
            volumes:
            networks:

Agregamos volumes


            version: "3"
            services:
              app:
                build: .
                ports:
                  - "3000:3000"
                  - "5000:5000"
                volumes:
                  - .:/app
                  - logvolume01:/var/log
                links:
                  - redis:redisAlias
              redis:
                image: redis
            volumes:
              logvolume01:
            networks:

Networking


            version: "3"
            services:
              app:
                build: .
                ports:
                  - "3000:3000"
                  - "5000:5000"
                volumes:
                  - .:/app
                  - logvolume01:/var/log
                links:
                  - redis:redisAlias
                networks:
                  - network1
              redis:
                image: redis
                networks:
                  - network1
            volumes:
              - logvolume01:
            networks:
              - network1:
                # Use a default or custom driver.
                - driver: bridge
